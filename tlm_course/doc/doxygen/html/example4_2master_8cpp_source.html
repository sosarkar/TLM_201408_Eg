38 void
39 Master::worker_thread()
40 {
41  // the data for the transaction
42  unsigned int data;
43 
44  // using single transaction throughout the process
45  tlm::tlm_generic_payload trans;
46 
47  // disable byte_enable
48  trans.set_byte_enable_ptr(0);
49 
50  // make the transaction use data for its data
51  trans.set_data_ptr((unsigned char *) &data);
52 
53  // set transaction length to 4 bytes = 1 integer
54  trans.set_streaming_width(4);
55  trans.set_data_length(4);
56 
57  sc_time local_time(50, SC_NS); // account for reset
58 
59  ostringstream msg;
60 
61  for (int ii =0; ii < 5; ii++) {
62  // wait 25 cycles
63  local_time += 25 * clock_period;
64 
65  // program a term
66  trans.set_write();
67  trans.set_dmi_allowed(false);
68  trans.set_response_status(tlm::TLM_INCOMPLETE_RESPONSE);
69  trans.set_address(REG_OFFSET_TERMS + (ii<<2));
70  data = (ii << 8) + 25;
71  init->b_transport(trans, local_time);
72 
73  msg.str("");
74  msg << name() << " @ " << sc_time_stamp() << "+" << local_time
75  << ": Transaction returned " << trans.get_response_string();
76  if (trans.is_response_error()) {
77  SC_REPORT_ERROR(__FILE__, msg.str().c_str());
78  }
79  SC_REPORT_INFO(__FILE__, msg.str().c_str());
80 
81  // wait 25 cycles
82  local_time += 25 * clock_period;
83 
84  // start the slave
85  trans.set_write();
86  trans.set_dmi_allowed(false);
87  trans.set_response_status(tlm::TLM_INCOMPLETE_RESPONSE);
88  trans.set_address(REG_START);
89  data = 0x1;
90  init->b_transport(trans, local_time);
91 
92  msg.str("");
93  msg << name() << " @ " << sc_time_stamp() << "+" << local_time
94  << ": Transaction returned " << trans.get_response_string();
95  if (trans.is_response_error()) {
96  SC_REPORT_ERROR(__FILE__, msg.str().c_str());
97  }
98  SC_REPORT_INFO(__FILE__, msg.str().c_str());
99 
100  wait(local_time);
101  wait(done.posedge_event());
102  // reading can start one clock after done
103  local_time = clock_period;
104 
105  // read the result
106  trans.set_read();
107  trans.set_dmi_allowed(false);
108  trans.set_response_status(tlm::TLM_INCOMPLETE_RESPONSE);
109  trans.set_address(REG_RESULT);
110  init->b_transport(trans, local_time);
111 
112  msg.str("");
113  msg << name() << " @ " << sc_time_stamp() << "+" << local_time
114  << ": Transaction returned " << trans.get_response_string();
115  if (trans.is_response_error()) {
116  SC_REPORT_ERROR(__FILE__, msg.str().c_str());
117  }
118  msg << ", result = " << data << " (0x" << hex << data << dec << ")";
119  SC_REPORT_INFO(__FILE__, msg.str().c_str());
120 
121  local_time += clock_period;
122  }
123  wait(local_time);
124 
125  msg.str("");
126  msg << name() << " @ " << sc_time_stamp() << ": end of simulation";
127  SC_REPORT_INFO(__FILE__, msg.str().c_str());
128 
129  sc_stop();
130 }
