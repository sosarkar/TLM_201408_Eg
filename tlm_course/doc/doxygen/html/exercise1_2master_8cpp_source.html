40 void
41 Master::ClockMethod()
42 {
43  if (resetn == false) {
44  start = false;
45  iterations_ = 0;
46  counter_ = 0;
47  slave_running_ = false;
48  } else {
49  if (start.read() != 0) {
50  // reset start signal after one cycle
51  start = 0;
52  } else if (iterations_ == 5) {
53  std::ostringstream msg;
54  msg << name() << " @ " << sc_time_stamp() << ": end of simulation";
55  SC_REPORT_INFO(__FILE__, msg.str().c_str());
56  sc_stop();
57  } else if (!slave_running_) {
58  // if the slave is not running, count till 50 and start
59  if (counter_ == 50) {
60  start = counter_.read() + (iterations_ << 8);
61  slave_running_ = true;
62  counter_ = 0;
63  } else {
64  counter_ = counter_ + 1;
65  }
66  } else if (done) {
67  // increment the iteration count if slave gives the done signal
68  iterations_ = iterations_ + 1 ;
69  slave_running_ = false;
70  // get and print result
71  sc_uint<16> result_local = result;
72  std::ostringstream msg;
73  msg << name() << " @ " << sc_time_stamp()
74  << ", Slave done with result " << result_local << " (0x"
75  << hex << result_local << ")" ;
76  SC_REPORT_INFO(__FILE__, msg.str().c_str());
77  } // the slave is running; wait until it is done
78  }
79 }
