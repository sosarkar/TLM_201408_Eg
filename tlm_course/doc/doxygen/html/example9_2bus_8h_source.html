85 template <unsigned int NR_OF_INITIATORS>
86 Bus<NR_OF_INITIATORS>::Bus(
87  sc_module_name name,
88  unsigned int bits_per_initiator,
89  unsigned int bits_extra):
90  sc_module(name),
91  target("target"),
92  bits_per_initiator_(bits_per_initiator),
93  mask_addr_((1<<bits_per_initiator)-1),
94  mask_((1 << (bits_extra + bits_per_initiator))-1)
95 {
96  if ((1 << bits_extra) < NR_OF_INITIATORS) {
97  ostringstream msg;
98  msg << this->name() << " @ " << sc_time_stamp()
99  << ": initiators outside bus memory range (" << NR_OF_INITIATORS
100  << " initiators, only " << (1 << bits_extra) << " addressable)";
101  SC_REPORT_WARNING(__FILE__, msg.str().c_str());
102  }
103  // binding the target port
104  target.bind(*this);
105 
106  // creating the initiator sockets, interfaces and binding them
107  for (int i = 0; i < NR_OF_INITIATORS; i++ ) {
108  initiators[i].register_nb_transport_bw(
109  this,
110  &Bus::my_nb_transport_bw,
111  i);
112  initiators[i].register_invalidate_direct_mem_ptr(
113  this,
114  &Bus::my_invalidate_direct_mem_ptr,
115  i);
116  }
117 }

134 template <unsigned int NR_OF_INITIATORS>
135 tlm_sync_enum
136 Bus<NR_OF_INITIATORS>::nb_transport_fw(
137  tlm_generic_payload &trans,
138  tlm_phase &phase,
139  sc_time &time )
140 {
141  tlm_sync_enum ret;
142  if (phase == BEGIN_REQ) {
143  unsigned int addr = trans.get_address() & mask_;
144  unsigned int length = trans.get_data_length();
145 
146  ostringstream msg;
147  msg << name() << " @ " << sc_time_stamp() << "+" << time << hex
148  << ": non-blocking forward transport, address 0x" << addr
149  << ", length 0x" << length << dec;
150 
151  //a transaction should go to a single target
152  unsigned int index = (addr >> bits_per_initiator_);
153  if (index >= NR_OF_INITIATORS) {
154  msg << ": address out of range";
155  SC_REPORT_WARNING(__FILE__, msg.str().c_str());
156  trans.set_response_status(TLM_ADDRESS_ERROR_RESPONSE);
157  ret = TLM_COMPLETED;
158  } else if (((addr + length-1) >> bits_per_initiator_ ) != index) {
159  msg << ": address out of range";
160  SC_REPORT_WARNING(__FILE__, msg.str().c_str());
161  trans.set_response_status(TLM_ADDRESS_ERROR_RESPONSE);
162  ret = TLM_COMPLETED;
163  } else {
164  msg << ": forwarding nb_transport_fw to initiator " << index;
165  SC_REPORT_INFO(__FILE__, msg.str().c_str());
166  trans.set_address(addr & mask_addr_);
167  ret= (initiators[index])->nb_transport_fw(trans, phase, time);
168  if ((ret == TLM_UPDATED) && (phase == BEGIN_RESP)) {
169  initiator_for_response_ = index;
170  }
171  }
172  } else {
173  sc_assert(phase == END_RESP);
174  ret= (initiators[initiator_for_response_])->nb_transport_fw(
175  trans,
176  phase,
177  time);
178  }
179  return ret;
180 }

214 template <unsigned int NR_OF_INITIATORS>
215 bool
216 Bus<NR_OF_INITIATORS>::get_direct_mem_ptr(
217  tlm_generic_payload &trans,
218  tlm_dmi &dmi_data)
219 {
220  unsigned int addr = trans.get_address() & mask_;
221  unsigned int length = trans.get_data_length();
222  bool ret;
223 
224  //a transaction should go to a single target
225  unsigned int index = (addr >> bits_per_initiator_);
226  if (((addr + length-1) >> bits_per_initiator_ ) != index) {
227  trans.set_response_status(TLM_ADDRESS_ERROR_RESPONSE);
228  ret = false;
229  } else {
230  trans.set_address(addr & mask_addr_);
231  ret = (initiators[index])->get_direct_mem_ptr(trans, dmi_data);
232  trans.set_address(addr);
233  }
234  return ret;
235 }
