49 void
50 Master::ClockMethod()
51 {
52  if (resetn == false) {
53  read_ena = false;
54  write_ena = false;
55  addr = 0;
56  write_data = 0;
57  iterations_ = 0;
58  counter_ = 0;
59  slave_running_ = false;
60  read_pending_ = false;
61  write_pending_ = false;
62  } else {
63  if (read_ena.read()) {
64  // clear the enable after one cycle
65  read_ena = false ;
66  } else if (write_ena) {
67  // clear the enable after one cycle
68  write_ena = false ;
69  }
70 
71  if (read_pending_) {
72  if (ready) {
73  // slave has processed the read, get the data
74  sc_uint<32> data_local = read_data;
75  sc_uint<32> addr_local = addr;
76 
77  std::ostringstream msg;
78  msg << name() << " @ " << sc_time_stamp() << ": read 0x" << hex
79  << data_local << " from address 0x" << addr_local;
80  SC_REPORT_INFO(__FILE__, msg.str().c_str());
81 
82  iterations_ = iterations_ + 1 ;
83  read_pending_ = false;
84  }
85  } else if (write_pending_) {
86  if (ready) {
87  // slave has processed the write
88  sc_uint<32> addr_local = addr;
89  if (addr_local == REG_START) {
90  slave_running_ = true;
91  counter_ = 0;
92  }
93  write_pending_ = false;
94  }
95  } else if (iterations_ == 5) {
96  std::ostringstream msg;
97  msg << name() << " @ " << sc_time_stamp() << ": end of simulation";
98  SC_REPORT_INFO(__FILE__, msg.str().c_str());
99  sc_stop();
100  } else if (!slave_running_) {
101  // if the slave is not running, count till 50 and start
102  counter_ = counter_ + 1;
103  if (counter_ == 50) {
104  write_ena = true;
105  addr = REG_START;
106  write_data = 0x1;
107  write_pending_ = true;
108  } else if (counter_ == 25) {
109  // for every iteration, at count 25, program a term
110  write_ena = true;
111  addr = REG_OFFSET_TERMS + (iterations_ << 2);
112  write_data = counter_ + (iterations_ << 8);
113  write_pending_ = true;
114  }
115  } else if (done) {
116  // start reading result
117  slave_running_ = false;
118  read_ena = true;
119  addr = REG_RESULT;
120  read_pending_ = true;
121  } // the slave is running; wait until it is done
122  }
123 }
