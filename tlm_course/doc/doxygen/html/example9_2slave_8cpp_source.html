60 Slave::my_run()
61 {
62  while (true) {
63  wait();
64  sum_ = 0;
65  done = false;
66  running_ = true;
67  wait(slave_clock_period * NR_TERMS);
68  sum_ = my_dsp_function(terms_, NR_TERMS);
69  running_ = false ;
70  done = true;
71  }

74 /* ------------------------------------------------------------------------- */
75 void
76 Slave::my_b_transport(tlm_generic_payload &trans, sc_time &local_time)
77 {
78  sc_assert((BUS_WIDTH % 32) == 0);
79  unsigned int length = trans.get_data_length();
80  unsigned int address = trans.get_address();
81 
82  //some checks and error responses to unsupported features
83  if (trans.get_byte_enable_ptr()){
84  // not supporting byte enable
85  trans.set_response_status(TLM_BYTE_ENABLE_ERROR_RESPONSE);
86  } else if (trans.get_streaming_width() < length) {
87  // not supporting streaming
88  trans.set_response_status(TLM_BURST_ERROR_RESPONSE);
89  } else if ((length%bus_width_in_bytes) || (address%bus_width_in_bytes)){
90  // not supporting unaligned access
91  trans.set_response_status(TLM_ADDRESS_ERROR_RESPONSE);
92  } else {
93  trans.set_response_status(TLM_OK_RESPONSE);
94  if (trans.is_read()) { // only allow reading from result register
95  if (address == REG_RESULT && (length == bus_width_in_bytes)) {
96  unsigned int *data_ptr = (unsigned int *) trans.get_data_ptr();
97  *data_ptr= sum_;
98  } else {
99  trans.set_response_status(TLM_ADDRESS_ERROR_RESPONSE);
100  }
101  local_time += (SLAVE_READ_DELAY+(length/bus_width_in_bytes))
102  * bus_clock_period;
103  } else {
104  // write
105  // only on the terms, burst access is taken into account
106  if ((address == REG_START) && (length == bus_width_in_bytes)) {
107  run_event_.notify(local_time+SLAVE_WRITE_DELAY*bus_clock_period);
108  } else if ((address < REG_OFFSET_TERMS)
109  || (address >= REG_OFFSET_TERMS+sizeof(int)*NR_TERMS)) {
110  trans.set_response_status(TLM_ADDRESS_ERROR_RESPONSE);
111  } else {
112  unsigned int index = (address - REG_OFFSET_TERMS)/sizeof(int);
113  unsigned int remaining = (NR_TERMS-index)*sizeof(int);
114  const void *data_ptr = (const void *) trans.get_data_ptr();
115  if (length<= remaining) {
116  memcpy((void *) &terms_[index], data_ptr, length);
117  } else {
118  memcpy((void *) &terms_[index], data_ptr, remaining);
119  }
120  }
121  local_time += (SLAVE_WRITE_DELAY+(length/bus_width_in_bytes))
122  * bus_clock_period;
123  }
124  }
125 }
